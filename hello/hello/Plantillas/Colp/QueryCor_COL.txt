select distinct
	trim(t.deudor_id) deudor_id,
	trim(upper(t.correo)) correo
from
	cbpo_colpatria.obligaciones_asignacion as oa
inner join cbpo_colpatria.obligaciones as o on
	o.obligacion_id = oa.obligacion_id
inner join cbpo_colpatria.correos as t on
	t.deudor_id = o.deudor_id 
where
	oa.asignacion_id = (
	select
		asignacion_id
	from
		cbpo_colpatria.asignaciones as a
	where
		estado is true)
order by trim(t.deudor_id);
