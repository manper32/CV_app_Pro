select distinct trim(deudor_id ) 
				,row_number ()
	 			OVER (PARTITION  BY	trim(deudor_id )
				ORDER BY	trim(deudor_id ) desc,
							correo desc) AS ROWNUMBER
				,correo --select *
from cbpo_codensa.correos
where correo != ''
order by trim(deudor_id )desc,row_number ()
	 			OVER (PARTITION  BY	trim(deudor_id )
				ORDER BY	trim(deudor_id) desc,
							correo desc);
